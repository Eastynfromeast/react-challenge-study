# Dynamic Metadata
    you can create dynamic metadata for [id] pages by creating a function 
    /(movie)/movies/[id]/page.tsx
    
    export async function generateMetadata({ params: { id } }: IParams) {
        const movie = await getMovie(id);
        return {
            title: movie.title,
        };
    }
