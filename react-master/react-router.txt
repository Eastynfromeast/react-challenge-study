    ## useNavigate
        When user logs in, you want to redirect
        or when user goes to some page but they are not authorized in that page
        ==> use useNavigate()

        in Header.tsx
        function Header() {
            const navigate = useNavigate();
            const onAboutClick = () => {
                navigate("/about");
            };
            return (
                <header>
                    <ul>
                        <li>
                            <Link to={"/"}>Home</Link>
                        </li>
                        <li>
                            <button onClick={onAboutClick}>About</button>
                        </li>
                    </ul>
                </header>
            );
        }
